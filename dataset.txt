	email
	password
	username
	role				0 - user, 99 - admin
	last_login
	remember_token
	created_at
	updated_at
	deleted_at
	
table : setting_defs		: 설정 항목 정의
	id
	setting_id
	title
	desc
	created_at
	updated_at
	
table : user_settings		: 사용자 별 설정 저장
	id
	user_id				0 - default value
	setting_id
	value
	created_at
	updated_at
	deleted_at

	context
	filename
	attach_key
	attach_type
	created_at
	updated_at
	deleted_at
